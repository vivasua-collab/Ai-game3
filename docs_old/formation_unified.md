### Внедрение формации в ядро:

```typescript
interface CoreImbuement {
  // === УСЛОВИЯ ===
  conditions: {
    // Практик знает формацию
    knowsFormation: boolean;
    // Формация совместима с ядром
    isCompatible: boolean;      // level формации ∈ core.levels
    // Ядро пустое (не занято другой формацией)
    isEmpty: boolean;
  };
  
  // === ПРОЦЕСС ===
  process: {
    // Практик тратит contourQi
    qiCost: number;             // contourQi формации
    // Время внедрения
    time: number;               // contourQi / (conductivity × qiDensity)
  };
  
  // === РЕЗУЛЬТАТ ===
  result: {
    // Ядро теперь содержит формацию
    coreFormation: FormationInCore;
    // Очистить ядро НЕЛЬЗЯ
    isPermanent: true;
  };
}

// Внедрение:
function imbueCoreWithFormation(
  core: FormationCore,
  formation: FormationKnowledge,
  practitioner: Character
): CoreImbuementResult {
  // Проверки
  if (!practitioner.knownFormations.includes(formation.id)) {
    return { success: false, error: 'Практик не знает эту формацию' };
  }
  
  if (!core.levels.includes(formation.level)) {
    return { success: false, error: 'Ядро не подходит для этого уровня' };
  }
  
  if (core.imbuedFormation) {
    return { success: false, error: 'Ядро уже содержит формацию' };
  }
  
  // Списание Ци
  const qiCost = formation.contourQi;
  if (practitioner.currentQi < qiCost) {
    return { success: false, error: 'Недостаточно Ци' };
  }
  
  practitioner.currentQi -= qiCost;
  
  // Внедрение
  core.imbuedFormation = {
    formationId: formation.id,
    level: formation.level,
    type: formation.type,
    size: formation.size,
    // Параметры копируются из формации
  };
  
  return {
    success: true,
    core,
    qiConsumed: qiCost,
  };
}
```

const CAPACITY_MULTIPLIER_BY_SIZE: Record<FormationSize, number> = {
  small: 10,      // ×10
  medium: 50,     // ×50

  great: 1000,    // ×1000

const HEAVY_CAPACITY_MULTIPLIER = 10000;

function calculateCapacity(
  level: number,
  size: FormationSize,
  isHeavy: boolean = false
): number {
  const contourQi = CONTOUR_QI_BY_LEVEL[level];
  
  if (isHeavy && level >= 6) {
    return contourQi * HEAVY_CAPACITY_MULTIPLIER;
  }
  
  return contourQi * CAPACITY_MULTIPLIER_BY_SIZE[size];
}

### Формулы:

```typescript
interface FormationFilling {
  formationId: string;
  
  // Участники
  participants: Array<{
    practitionerId: string;
    isCreator: boolean;
    conductivity: number;
    qiDensity: number;
    contribution: number;      // Сколько уже влил
  }>;
  
  // Текущее состояние
  currentQi: number;
  capacity: number;
  fillPercent: number;
  
  // Скорость наполнения
  totalFillRate: number;
}

// Расчёт скорости (простая сумма):
function calculateFillRate(filling: FormationFilling): number {
  return filling.participants.reduce((sum, p) => 
    sum + (p.conductivity * p.qiDensity), 0
  );
}

// Время наполнения:
function calculateFillTime(
  capacity: number,
  fillRate: number
): number {
  return capacity / fillRate;  // секунд
}

// Пример: L5 Barrier Great
// capacity = 1,280,000 Ци
// L5 практик (cond=8, dens=16): 128 Ци/сек
// 5 практиков L5: 128 × 5 = 640 Ци/сек
// Время: 1,280,000 / 640 = 2000 сек ≈ 33 минуты
```

## 🛡️ 1. БАРЬЕРНЫЕ ФОРМАЦИИ (Barrier)

### Структура:

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                     БАРЬЕРНАЯ ФОРМАЦИЯ                                       │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  ПАРАМЕТРЫ:                                                                  │
│  ─────────────────────────────────────────────────────────────────────────  │
│  - contourQi: стоимость прорисовки контура                                  │
│  - capacity: ёмкость аккумулятора                                           │
│  - currentQi: текущий заряд                                                 │
│  - qiCostRatio: коэффициент затрат Ци на урон                               │
│  - criticalThreshold: 0 (схлопывается только при полной потере)            │
│                                                                              │
│  ПРИНЦИП РАБОТЫ:                                                            │
│  ─────────────────────────────────────────────────────────────────────────  │
│  1. Поглощает 100% урона пока есть заряд                                    │
│  2. Урон списывает Ци из аккумулятора                                       │
│  3. Коэффициент затрат зависит от уровня:                                   │
│     • L1: 1 урон = 1.0 Ци                                                   │
│     • L5: 1 урон = 0.5 Ци                                                   │
│     • L9: 1 урон = 0.25 Ци                                                  │
│  4. При 0 Ци — формация схлопывается                                        │
│  5. Естественная утечка Ци со временем                                      │

│  - Защита городов                                                            │
│  - Защита сект                                                               │

│                                                                              │

│  - contourQi: базовый (ядро)                                                │
│  + contourQi × 0.5 × linesCount (линии подпитки)                            │
│  + contourQi × 0.001 × perimeterLength (периметр)                           │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### Расчёт стоимости:

```typescript
function calculateHeavyFormationCost(
  level: number,
  perimeterLength: number,
  supplyLineCount: number
): number {
  const base = CONTOUR_QI_BY_LEVEL[level];
  
  const coreCost = base;
  const linesCost = base * 0.5 * supplyLineCount;
  const perimeterCost = base * 0.001 * perimeterLength;
  
  return coreCost + linesCost + perimeterCost;
}

// Пример L8, периметр 15 км, 8 линий:
// base = 10240
// coreCost = 10240
// linesCost = 10240 × 0.5 × 8 = 40960
// perimeterCost = 10240 × 0.001 × 15000 = 153600
// ИТОГО: 204800 Ци на прорисовку
```
