## 📐 ФОРМУЛЫ

### Стоимость активации (contourQi):

```typescript
// contourQi — Ци на прорисовку контура
// Выполняет ТОЛЬКО ОДИН практик (создатель)

const CONTOUR_QI_BY_LEVEL: Record<number, number> = {
  1: 80,       // 80 × 2^0
  2: 160,
  3: 320,
  4: 640,
  5: 1280,
  6: 2560,
  7: 5120,
  8: 10240,
  9: 20480,
};

// Время прорисовки:
const drawTime = contourQi / (conductivity * qiDensity); // секунды
```

### Ёмкость формации (capacity):

```typescript
// НОВЫЕ МНОЖИТЕЛИ (v4):
const CAPACITY_MULTIPLIER_BY_SIZE: Record<FormationSize, number> = {
  small: 10,      // ×10 — минимум в 10 раз больше активации
  medium: 50,     // ×50
  large: 200,     // ×200
  great: 1000,    // ×1000 — минимум в 1000 раз
};

// Тяжёлые формации (L6+):
const HEAVY_CAPACITY_MULTIPLIER = 10000;  // ×10000 — защита городов

// Расчёт ёмкости:
function calculateCapacity(level: number, size: FormationSize, isHeavy = false): number {
  const contourQi = CONTOUR_QI_BY_LEVEL[level];
  if (isHeavy && level >= 6) {
    return contourQi * HEAVY_CAPACITY_MULTIPLIER;
  }
  return contourQi * CAPACITY_MULTIPLIER_BY_SIZE[size];
}

// Примеры:
// L1 Small:  80 × 10 = 800
// L5 Great:  1280 × 1000 = 1,280,000
// L8 Heavy:  10240 × 10000 = 102,400,000
```

### Механика (ОБНОВЛЕНО v4):

```typescript
interface FormationFilling {
  formationId: string;
  
  // Создатель (прорисовывал контур / внедрял в ядро)
  creatorId: string;
  
  // Участники (помогают вливать Ци)
  participants: Array<{
    practitionerId: string;
    isCreator: boolean;
    conductivity: number;
    qiDensity: number;
    contribution: number;      // Сколько уже влил
  }>;
  
  // Текущее состояние
  currentQi: number;
  capacity: number;
  fillPercent: number;
}

// Расчёт скорости наполнения:
function calculateFillRate(filling: FormationFilling): number {
  // Каждый участник вносит 100% от своей способности
  // Никаких бонусов/штрафов — просто сумма
  return filling.participants.reduce((sum, p) => 
    sum + (p.conductivity * p.qiDensity), 0
  );
}

// Время наполнения:
const fillTime = capacity / totalFillRate; // секунд
```

## 🔧 ИНТЕРФЕЙСЫ (ОБНОВЛЕНО v4)

```typescript
interface FormationState {
  id: string;
  techniqueId: string;
  
  // Этап жизни
  stage: 'drawing' | 'imbuing' | 'mounting' | 'filling' | 'active' | 'depleted';
  
  // Контур
  contourQi: number;         // Затрачено на контур
  drawTime: number;          // Время прорисовки (сек)
  
  // Ядро (опционально)
  hasCore: boolean;
  coreType?: 'disk' | 'altar';
  coreId?: string;
  
  // Наполнение
  currentQi: number;         // Текущее Ци
  maxCapacity: number;       // Максимум
  
  // Участники
  creatorId: string;
  participants: FormationParticipant[];
  totalFillRate: number;
  
  // Параметры
  type: FormationType;
  size: FormationSize;
  level: number;
  
  // Радиусы
  creationRadius: number;    // Радиус прорисовки
  effectRadius: number;      // Радиус действия
  
  // Естественная утечка (по formation_drain_system.md)
  drainInterval: number;     // Интервал в тиках
  drainAmount: number;       // Ци за раз
  drainPerHour: number;      // Ци/час
  lastDrainTick: number;     // Последний тик утечки
}

interface FormationParticipant {
  practitionerId: string;
  isCreator: boolean;
  conductivity: number;
  qiDensity: number;
  contribution: number;      // Сколько влил
  // NOTE: Каждый вносит 100%, никаких efficiency коэффициентов
}

// Проверка активации:
function canActivate(formation: FormationState): boolean {
  return formation.currentQi >= formation.maxCapacity;  // 100%
}
```

```
│  - Требует периодической замены камней                                      │
│                                                                              │
│  2. КОНТУР СБОРА Ци (L8+)                                                   │
│  ─────────────────────────────────────────────────────────────────────────  │
│  - Встроенный контур, собирающий Ци из среды                                │
│  - Эффективность зависит от плотности Ци в месте                            │
│  - baseRate: 2 ед/сек × environmentMultiplier                               │
│                                                                              │
│  3. ЛЕЙ-ЛИНИИ (географические потоки)                                       │
│  ─────────────────────────────────────────────────────────────────────────  │
│  - Формация на пересечении потоков Ци                                       │

│                                                                              │
│  4. ПРАКТИКИ (ручная подпитка)                                              │
│  ─────────────────────────────────────────────────────────────────────────  │
│  - Несколько практиков постоянно вливают Ци                                 │
│  - Каждый вносит 100% от способности                                        │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### Контур сбора Ци (L8+):

```typescript
const ENVIRONMENT_MULTIPLIER = {
  barren: 0.1,      // Пустыня
  normal: 0.5,      // Обычная местность
  rich: 1.0,        // Богатое Ци место
  spiritual: 2.0,   // Духовная земля секты
  sacred: 5.0,      // Священное место
};

function calculateQiCollectionRate(
  formationLevel: number,
  environment: keyof typeof ENVIRONMENT_MULTIPLIER
): number {
  if (formationLevel < 8) return 0;
  
  const baseRate = 2;  // ед/сек
  return baseRate * ENVIRONMENT_MULTIPLIER[environment] * (formationLevel / 8);
}

// Примеры:
// L8 в духовной земле: 2 × 2.0 × 1 = 4 ед/сек = 14,400 Ци/час
// L9 в священном месте: 2 × 5.0 × 1.125 = 11.25 ед/сек = 40,500 Ци/час
```
